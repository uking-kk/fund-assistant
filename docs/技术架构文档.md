## 三、核心模块设计

### 3.1 Agent 主逻辑

```python
# backend/app/core/agent.py

from langchain.agents import AgentExecutor
from langchain.tools import Tool
from app.core.rag import RAGRetriever
from app.services.fund_api import FundAPI
from app.services.llm import get_llm

class FundAssistantAgent:
    """
    基金助手Agent主类
    
    工作流程:
    1. 接收用户问题
    2. 意图识别（问答/查询/推荐）
    3. 调用对应工具
    4. LLM生成回答
    """
    
    def __init__(self):
        self.llm = get_llm()
        self.rag = RAGRetriever()
        self.fund_api = FundAPI()
        self.tools = self._init_tools()
        
    def _init_tools(self):
        """初始化工具集"""
        return [
            Tool(
                name="knowledge_qa",
                description="回答基金基础知识问题，如'什么是ETF'",
                func=self._knowledge_qa
            ),
            Tool(
                name="fund_query",
                description="查询基金数据，如'查一下易方达蓝筹精选'",
                func=self._fund_query
            ),
            Tool(
                name="fund_recommend",
                description="推荐基金，如'推荐稳健的债券基金'",
                func=self._fund_recommend
            )
        ]
    
    async def chat(self, question: str) -> AsyncGenerator[str, None]:
        """
        对话主入口 - 流式输出
        
        Args:
            question: 用户问题
            
        Yields:
            str: 流式输出的回答片段
        """
        # 1. 意图识别
        intent = await self._detect_intent(question)
        
        # 2. 根据意图调用不同工具
        if intent == "query":
            context = await self._fund_query(question)
        elif intent == "recommend":
            context = await self._fund_recommend(question)
        else:
            context = await self._knowledge_qa(question)
        
        # 3. LLM生成回答（流式）
        async for chunk in self._generate_response(question, context):
            yield chunk
```

### 3.2 RAG 检索模块

```python
# backend/app/core/rag.py

from langchain.vectorstores import Chroma
from langchain.embeddings import HuggingFaceEmbeddings
from langchain.text_splitter import RecursiveCharacterTextSplitter

class RAGRetriever:
    """
    RAG检索模块
    
    工作流程:
    1. 加载知识库文档
    2. 文本分块
    3. 向量化存储
    4. 相似度检索
    """
    
    def __init__(self, persist_directory: str = "./chroma_db"):
        # 使用BGE中文向量模型
        self.embeddings = HuggingFaceEmbeddings(
            model_name="BAAI/bge-small-zh-v1.5"
        )
        
        # 初始化向量数据库
        self.vectorstore = Chroma(
            persist_directory=persist_directory,
            embedding_function=self.embeddings
        )
        
        # 文本分块器
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=500,      # 每块500字符
            chunk_overlap=50,    # 重叠50字符
            separators=["\n\n", "\n", "。", "，"]
        )
    
    def add_documents(self, documents: list[str]):
        """
        添加文档到知识库
        
        Args:
            documents: 文档列表
        """
        # 分块
        chunks = self.text_splitter.split_text("\n".join(documents))
        
        # 向量化存储
        self.vectorstore.add_texts(chunks)
        self.vectorstore.persist()
    
    def retrieve(self, query: str, k: int = 3) -> list[str]:
        """
        检索相关文档
        
        Args:
            query: 查询文本
            k: 返回top-k个结果
            
        Returns:
            相关文档列表
        """
        results = self.vectorstore.similarity_search(query, k=k)
        return [doc.page_content for doc in results]
```

### 3.3 基金API服务

```python
# backend/app/services/fund_api.py

import httpx
from typing import Optional

class FundAPI:
    """
    天天基金API封装
    
    API文档: https://api.doctorxiong.club/
    """
    
    BASE_URL = "https://api.doctorxiong.club/v1"
    
    async def search_fund(self, keyword: str) -> list[dict]:
        """
        搜索基金
        
        Args:
            keyword: 基金名称或代码
            
        Returns:
            基金列表
        """
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{self.BASE_URL}/fund/search",
                json={"keyword": keyword}
            )
            return response.json().get("data", [])
    
    async def get_fund_detail(self, fund_code: str) -> dict:
        """
        获取基金详情
        
        Args:
            fund_code: 基金代码
            
        Returns:
            基金详情信息
        """
        async with httpx.AsyncClient() as client:
            response = await client.post(
                f"{self.BASE_URL}/fund/detail",
                json={"fundCode": fund_code}
            )
            data = response.json().get("data", {})
            
            # 格式化返回数据
            return {
                "code": fund_code,
                "name": data.get("name"),
                "type": data.get("fundType"),
                "manager": data.get("managerName"),
                "scale": data.get("fundScale"),
                "returns": {
                    "1y": data.get("returnRate1y"),
                    "3y": data.get("returnRate3y"),
                },
                "risk_level": data.get("riskLevel")
            }
```

```jsx
import { useState, useRef, useEffect } from 'react'
import { motion, AnimatePresence } from 'framer-motion'
import { useChat } from '@/hooks/useChat'
import MessageList from './MessageList'
import FundCard from './FundCard'

export default function ChatWindow() {
  const [input, setInput] = useState('')
  const { messages, sendMessage, isLoading } = useChat()
  const messagesEndRef = useRef(null)

  // 自动滚动到底部
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' })

    setInput('')

      {/* 消息列表 */}
```

### 5.2 对话Hook

```jsx
// frontend/src/hooks/useChat.js

import { useState, useCallback } from 'react'
import { chatApi } from '@/lib/api'

export function useChat() {
  const [messages, setMessages] = useState([
    {
      role: 'assistant',
      content: '你好！我是基金智能助手，有什么可以帮你的？'
    }
  ])
  const [isLoading, setIsLoading] = useState(false)

  const sendMessage = useCallback(async (content) => {
    // 添加用户消息
    setMessages(prev => [...prev, { role: 'user', content }])
    setIsLoading(true)

    try {
      // 流式接收AI回复
      let aiMessage = ''
      await chatApi.streamChat(content, (chunk) => {
        aiMessage += chunk
        // 实时更新AI消息
        setMessages(prev => {
          const newMessages = [...prev]
          const lastIndex = newMessages.length - 1
          if (newMessages[lastIndex]?.role === 'assistant') {
            newMessages[lastIndex].content = aiMessage
          } else {
            newMessages.push({ role: 'assistant', content: aiMessage })
          }
          return newMessages
        })
      })
    } finally {
      setIsLoading(false)
    }
  }, [])

  return { messages, sendMessage, isLoading }
}
```
